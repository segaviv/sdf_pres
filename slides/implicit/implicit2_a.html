Code example for drawing a circle


    #define RADIUS 0.5
    #define RED vec3(0.8, 0.1, 0.3)
    #define WHITE vec3(1.0)
    float sdCirc(vec2 p) {
      return length(p) - RADIUS;
     }
    void mainImage(out vec4 gl_FragColor, in vec2 p) {
      gl_FragColor = length(p) - RADIUS < 0.
                        ? RED : WHITE;
    }
