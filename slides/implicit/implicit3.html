Example of other implicit functions


      float sdBox(vec2 p, vec2 b) {
        vec2 d = abs(p)-b;
        return length(max(d,0.0)) + 
               min(max(d.x,d.y),0.0);
      }
    

      float sdHeart(vec2 p, vec2 a, vec2 b) {
        p.x = abs(p.x);
        return p.y+p.x>1.0 ? sqrt(dot2(p-a))-sqrt(2.0)/4.0
        : sqrt(min(dot2(p-b), dot2(p-0.5*max(p.x+p.y,0.0)))) * sign(p.x-p.y);
      }
    

      float sdBox(vec2 p, vec2 b) {
        vec2 d = abs(p)-b;
        return length(max(d,0.0)) + 
               min(max(d.x,d.y),0.0);
      }
